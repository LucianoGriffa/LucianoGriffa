def save_profile(new_profile):
    profiles = []
    if new_profile:
        profiles.append(new_profile)
        print("Profile saved successfully:", profiles)
    else:
        print("No data to save.")

    "name": "Luciano",
    "student": "📚 Cybersecurity",
    "enthusiast": ["Technology", "Cybersecurity", "Music"]
}
