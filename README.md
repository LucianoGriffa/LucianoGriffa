const saveProfile = (newProfile) => {
  const profiles = [];
  if (newProfile) {

  }
};

  name: "Luciano",
  role: "💻 Full Stack Developer",
  student: "📚 Software Engineer",
  enthusiast: ["Technology", "Music"]
};
saveProfile(myProfile)
